# 常用mixin

// flex布局
@mixin flex($j: center, $a: center, $d: row, $im: false) {
    @if ($im) {
        display: flex;
        justify-content: $j !important;
        align-items: $a !important;
        flex-direction: $d !important;
    } @else {
        display: flex;
        flex-direction: $d;
        justify-content: $j;
        align-items: $a;
    }
}

// 单行多行文本溢出
@mixin ellipsis($isMore: false,$line:3) {
    @if ($isMore) {
        display: -webkit-box;
        -webkit-box-orient: horizontal;
        -webkit-line-clamp: $line;
        overflow: hidden;
    } @else {
        overflow: hidden;
        text-overflow: ellipsis;
        white-space: nowrap;
    }
}

//绝对定位
@mixin absolute($t: 0, $r: 0, $b: 0, $l: 0) {
    position: absolute;
    top: $t;
    bottom: $b;
    left: $l;
    right: $r;
}

 @mixin clearfix {
  &:after {
    content: "";
    display: table;
    clear: both;
  }
}

@mixin scrollBar($w: 10px) {
  &::-webkit-scrollbar-track-piece {
    background: transparent;
  }

  &::-webkit-scrollbar {
    width: $w;
  }

  &::-webkit-scrollbar-thumb {
    background: #dddddd;
    // border-radius: 20px;
  }
  scrollbar-width: thin;
  scrollbar-color: transparent;
}


@mixin triangle($width, $height, $color, $direction) {
  $width: $width/2;
  $color-border-style: $height solid $color;
  $transparent-border-style: $width solid transparent;
  height: 0;
  width: 0;

  @if $direction==up {
    border-bottom: $color-border-style;
    border-left: $transparent-border-style;
    border-right: $transparent-border-style;
  } @else if $direction==right {
    border-left: $color-border-style;
    border-top: $transparent-border-style;
    border-bottom: $transparent-border-style;
  } @else if $direction==down {
    border-top: $color-border-style;
    border-left: $transparent-border-style;
    border-right: $transparent-border-style;
  } @else if $direction==left {
    border-right: $color-border-style;
    border-top: $transparent-border-style;
    border-bottom: $transparent-border-style;
  }
}
