# ECMAScript历代版本特性 - ES9(2018)

# rest/spread 属性

# 基本合并*

通过向对象文法添加 spread 属性，ES2018 进一步扩展了此语法。 使用 spread 属性，你可以将对象自身的可枚举属性复制到新对象上。

// 对象合并
const a = {a:1,b:2};
      b = {b:231,d:11};
console.log({...a,...b}); //{a: 1, b: 231, d: 11}

// 数组合并
const a =[1,2,4],
      b =[1,2,6];

console.log([...a,...b]); //[1, 2, 4, 1, 2, 6]

# spread 属性只复制可枚举属性

const car ={name:"benz"};
Object.defineProperty(car,'color',{
    value:"black",
    enumerable:false,
});

console.log({...car});  //{name: 'benz'};

const car ={name:"benz"};
Object.defineProperty(car,'color',{
    value:"white",
    enumerable:true,
});
console.log({...car});  //{name: 'benz', color: 'white'}

# 结合解构赋值

const user ={name:"行政区",value:"120km",age:"1440years"};

let {name,...rest} = user;

console.log(rest);  //{value: '120km', age: '1440years'}

# async/await 结合for..of..异步迭代

迭代数据集是编程的重要组成部分。 在 ES2015 之前，JavaScript提供了for,for...in 和 while 等语句，以及map(),filter() 和 forEach() 等方法。 为了方便程序员一个个地处理集合元素，ES2015 引入了迭代器接口。

如果对象具有Symbol.iterator属性，则表示它是可迭代的。 在 ES2015 中，字符串和集合对象（如Set, Map 和 Array）带有Symbol.iterator 属性，因此是可迭代的。object是不支持Symbol.iterator属性的 Symbol.iterator 是个广为人知的符号，用于表示返回迭代器的函数。与迭代器交互主要使用next()方法。此方法返回一个具有value和done 两个属性的对象。value属性包含集合中下一个元素的值。done 属性包含true或false,表明是否已到达集合的末尾。

// Symbol.iterator 示例
const arr =[1,2,4],
     iterator = arr[Symbol.iterator]();
console.log(iterator.next()); //{value: 1, done: false}
console.log(iterator.next()); // {value: 2, done: false}
console.log(iterator.next()); // {value: 4, done: false}
console.log(iterator.next()); // {value: undefined, done: true}

自定义Symbol.iterator属性使object也支持**next()**的方法：

const user = {name:"xiaolei", age:"100", sex: "gents", 
        [Symbol.iterator]:function * (){
            for(var key in this){
                yield this[key]; //yield是ES6中的Generator（生成器）用来内部暂停本身运行的关键字。
            }
            return 'loop finished';
        }},
      iterator = user[Symbol.iterator]();

console.log(iterator.next()); // {value: 'xiaolei', done: false}
console.log(iterator.next()); // {value: '100', done: false}
console.log(iterator.next()); // {value: 'gents', done: false}
console.log(iterator.next()); // {value: 'loop finished', done: true}

使用 for...of 语句可以轻松迭代可迭代对象，但是 for...of 不能与异步可迭代对象一起使用，因为value 和 done不是同步产生的。 出于这个原因，ES2018 提供了 for...await...of语句。

const user = {name:"xiaolei", age:"100", sex: "gents", 
        [Symbol.asyncIterator]:async function * (){
            for(var key in this){
                yield this[key]; //yield是ES6中的Generator（生成器）用来内部暂停本身运行的关键字。
            }
            return 'loop finished';
        }};
      
(async ()=>{
    for await(const res of user){
        console.log(res);
    }
})();

// xiaolei
// 100
// gents
// Promise {<fulfilled>: undefined}

for...await...of 语句隐式调用集合对象上的Symbol.asyncIterator方法以获取异步迭代器。 每次循环时，都会调用迭代器的next()方法，该方法返回一个 promise. 一旦 promise 完成，就会将结果对象的 value 属性读取到x变量。 循环继续，直到返回对象的done 属性值为true.

# 异步操作循环中的串行

const a =[1,2,4,5],
     request = async function(){
       return await fetch('https://www.vicray.cn:3886/test').then(res => res.json())

const loopTest = async function(){
    console.time('serial')
    for (const number of a){

# 异步操作循环的并行执行

const a = [1,2,3,4,5,6,7],
      request = async function(){
       return await fetch('https://www.vicray.cn:3886/test').then(res => res.json())
      };
const Parallelism = function(){
    console.time('Parallelism');
    Promise.all(a.map(async t => {
            const res = await request();
            console.log('第'+t+"次之后后结果是"+JSON.stringify(res));
    }))
    console.timeEnd('Parallelism');
}

Parallelism();  //Parallelism: 2.86181640625 ms

# Promise.finally()

finally() 方法可在操作完成后进行一些扫尾（clean up）工作，无论操作是否成功。 在此代码中，finally() 方法在数据获取处理后直接隐藏了加载 spinner. 无论 promise 完成与否，函数中的注册代码都会执行，开发者不必在 then() 和 catch()方法中重复编写逻辑。

new Promise()
    .then()
    .catch(e => e)
    .finally();

# 正则表达式新特性

# s（dotAll）标志

s 标志是 ‘single line'(单行)的缩写，它使 . 匹配新的行字符。如果没有它，点将匹配普通字符，而不是新行

/hi.welcome/.test('hi\nwelcome') // false
/hi.welcome/s.test('hi\nwelcome') // true

# 可命名捕获组

const re = /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/
const result = re.exec('2015-01-02')
// result.groups.year === '2015';
// result.groups.month === '01';
// result.groups.day === '02';

# 先行断言(lookahead)：

您可以使用 ?= 匹配一个字符串，该字符串后面紧跟着一个特定的子字符串：

/Roger(?=Waters)/ //正向先行断言

/Roger(?= Waters)/.test('Roger is my dog') //false
/Roger(?= Waters)/g.test('Roger is my dog and Rogers Waters is a famous musician') // false
/Roger(?= Waters)/.test('Roger is my dog and Roger Waters is a famous musician') // true

?! 执行逆操作，匹配一个字符串，紧跟该字符串后面没有一个特定的子字符串：

/Roger(?!Waters)/ // 负向先行断言
 
/Roger(?! Waters)/.test('Roger is my dog') //true
/Roger(?! Waters)/.test('Roger Waters is a famous musician') //false
/Roger(?! Waters)/.test('Roger is a famous musician, and his lastName is Waters ') //true

(?<=pattern): 正向后行断言后行断言(lookbehind)，是一个新功能，使用 ?<=。

/(?<=Roger) Waters/  ///正向后行断言
 
/(?<=Roger) Waters/.test('Pink Waters is my dog') //false
/(?<=Roger) Waters/.test('Roger is my dog and Roger Waters is a famous musician')

(?<!pattern)： 后行断言(lookbehind) 逆操作，使用 ?<= !。

代表字符串中的一个位置，紧接该位置之前的字符序列不能匹配 pattern。

/(?<!Roger) Waters/  //负向后行断言
  
/(?<!Roger) Waters/.test('Pink Waters is my dog') //true
/(?<!Roger) Waters/.test('Roger is my dog and Roger Waters is a famous musician'); //false
/Waters (?<!Roger) [0-9]/.test('Roger is my dog and Roger Waters is a famous musician'); //true

# Unicode 属性转义 \p{…} 和 \P{…}

这个新功能将扩展此概念到引入 \p{} 匹配所有 Unicode 字符，否定为 \P{}.任何 unicode 字符都有一组属性。 例如，Script 确定语言系列，ASCII 是一个布尔值， 对于 ASCII 字符，值为 true，依此类推。 您可以将此属性放在花括号中，正则表达式将检查是否为真：

/^\p{ASCII}+$/u.test('abc')   //✅
/^\p{ASCII}+$/u.test('ABC@')  //✅
/^\p{ASCII}+$/u.test('ABC🙃') //❌

ASCII_Hex_Digit是另一个布尔属性，用于检查字符串是否仅包含有效的十六进制数字：

/^\p{ASCII_Hex_Digit}+$/u.test('g'); //false
/^\p{ASCII_Hex_Digit}+$/u.test('A'); //true
/^\p{ASCII_Hex_Digit}+$/u.test('ade3314124fe'); //true

还有许多其他布尔属性，您只需通过在花括号中添加它们的名称来检查它们，包括 Uppercase, Lowercase, White_Space, Alphabetic, Emoji 等：

/^\p{Lowercase}$/u.test('h') //true
/^\p{Uppercase}$/u.test('H') //true
/^\p{Emoji}+$/u.test('H')   // false
/^\p{Emoji}+$/u.test('🙃🙃') //✅
