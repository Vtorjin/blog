# 算法系列之: 罗马字符转为数字

题目要求：给定一个罗马数字,将其转为整数。其中罗马数字包含以下七种字符: I， V， X， L，C，D 和 M。

字符          数值
I             1
V             5
X             10
L             50
C             100
D             500
M             1000

例如， 罗马数字 2 写做 II ，即为两个并列的 1 。12 写做 XII ，即为 X + II 。 27 写做  XXVII, 即为 XX + V + II 。

通常情况下，罗马数字中小的数字在大的数字的右边。但也存在特例，例如 4 不写做 IIII，而是 IV。数字 1 在数字 5 的左边，所表示的数等于大数 5 减小数 1 得到的数值 4 。同样地，数字 9 表示为 IX。这个特殊的规则只适用于以下六种情况：

I 可以放在 V (5) 和 X (10) 的左边，来表示 4 和 9。 X 可以放在 L (50) 和 C (100) 的左边，来表示 40 和 90。  C 可以放在 D (500) 和 M (1000) 的左边，来表示 400 和 900。 给定一个罗马数字，将其转换成整数。

比如: 输入: s = "III" 输出: 3

输入: s = "LVIII" 输出: 58 解释: L = 50, V= 5, III = 3.

个人思路: 1.一般来说罗马数字中小的数字在大的数字的右边,但是有6种情况是特殊的,小的数值是在大的数字左边,并且6种值的展现形式都有一个共同点(较小的值和最大的值组合在一起时的差距就是一个最小值(比如4的展示形式是(1和5),90的展示形式是(10和100)))，其余的形式就是通过不断的累加所得到的。 2.需要有一个存放这7种对应的标记,以及返回结果的变量 3.对传入的字符进行遍历,如果当前一位的数值大于后面的一位数值，就进行累加，小于就减去当前位的值. 4.

# 个人代码实现(不考虑代码输入的大小写或者不满足输入格式的数)

const romanToInt =(str)=>{
    let result=0;
    // 标记
    let map = {
        I: 1,
        V: 5,
        X: 10,
        L: 50,
        C: 100,
        D: 500,
        M: 1000
    };
    let i = 0,len = str.length;
    
    for(i;i<len;i++){
        //判断当前一位和下一位的数值的大小关系
        if(map[str[i]]<map[str[i+1]]){
            result -= map[str[i]];
        }else{
            result += map[str[i]];
        }
    }
    return result;
}

# 其他代码实现

var  romanToInt = function(s) {
    let luoma_obj = {"I": 1, "V": 5, "X": 10, "L": 50, "C": 100, "D": 500, "M": 1000}
    let index = 0
    let sum = 0

    while (true) {
        if (index >= s.length) {
            break
        }

        if (index < s.length - 1) {
            switch (s[index]) {
                case "I":
                    switch (s[index + 1]) {
                        case "V":
                            sum = sum + 4;
                            index += 1;
                            break
                        case "X":
                            sum = sum + 9;
                            index += 1;
                            break
                        default:
                            sum = sum + 1;
                            break
                    }
                    break
                case "X":
                    switch (s[index + 1]) {
                        case "L":
                            sum = sum + 40;
                            index += 1;
                            break
                        case "C":
                            sum = sum + 90;
                            index += 1;
                            break
                        default:
                            sum = sum + 10;
                            break
                    }
                    break
                case "C":
                    switch (s[index + 1]) {
                        case "D":
                            sum = sum + 400;
                            index += 1;
                            break
                        case "M":
                            sum = sum + 900;
                            index += 1;
                            break
                        default:
                            sum = sum + 100;
                            break
                    }
                    break
                default:
                    sum = sum + luoma_obj[s[index]]
            }
        } else {
            sum = sum + luoma_obj[s[index]]
        }

        index += 1;
    }
    return sum
};

思路简析:

  1. 需要有一个存放这7种对应的标记,当前下标,以及返回结果的变量
  2. 对6种特殊情况进行判断,进行特殊值赋值处理.
  3. 对传入的数据进行从左到右循环,通过比较当前位和下一位的数值进行+或者特殊值的处理。
