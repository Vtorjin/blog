# js常用功能- FormData&Browser

# 将一组表单元素转为为对象

const formDataToObject = form  =>{
  Array.from(new FormData(form)).reduce((acc, [key, value]) => ({
      ...acc,
      [key]: value
    }), {});
}  

formDataToObject(document.querySelector('#form')); // { email: 'test@email.com', name: 'Test Name' }

# object数据格式转换为formData对象


const objectToFormData = objData => {
    const form = new FormData();
    Object.keys(objData).forEach(key=>{
        form.append(key, Object[key]);
    })
    return form;
}

objectToFormData({username:"vicray",pwd:"123456"});

# 获取get请求的参数对象指定key的值

const getUrlParam = (sUrl, sKey) => {
    const index = sUrl.indexOf('?');
    const urlStr = sUrl.substr(index+1);
    const reg = /&?([^=]+)=([^&#]+)/g;
    const result = {};
    urlStr.replace(reg,function(re,$1,$2){
        if(result[$1]){
            var item = result[$1];
            result[$1] = [].concat(item,$2);
        }else{
            result[$1] = $2;
        }
    });

    return sKey ? (result[sKey] || ''): '' ;
}

# 获取当前域名相关信息

//@param {type:string}   //可用参数:['protocol','host','href','hostname','pathname','port']
const getHostInfo = (type = 'all') => {
    if(!['all','protocol','host','href','hostname','pathname','port'].includes(type)) return '';
    return  type === 'all' ? window.location.href : window.location[type]
}

注: 获取的域名信息参数有以下几种:

  • all:获取协议和域名的组合信息(包含端口);
  • protocol: 获取协议信息(https or http);
  • host: 获取主机信息(包含端口);
  • href: 获取完整的路径(包含路径和请求(get)参数);
  • hostname: 获取主机信息(不包含端口);
  • pathname: 获取请求地址
  • port: 获取端口号
