# 2.数组的交集,并集,差集(es5 Array)

var A = [1, 2, 3, 5, 7];
var B = [2, 4, 5, 6, 7];

// 交集
var intersection = A.filter(function (v) {
    return B.includes(v);
}) //[2, 5, 7]

// 差集
var diff = A.filter(function (v) {
    return !B.includes(v);
}) // [1, 3]

// 并集
var union = B.concat(A.filter(function (v) {
    return !B.includes(v);
})) // [2, 4, 5, 6, 7, 1, 3]

//补集
var complement = (function () {
    var complementA = A.filter(function (v) { return !B.includes(v) })
    var complementB = B.filter(function (v) { return !A.includes(v) })
    return complementA.concat(complementB);
})() //[1, 3, 4, 6]

# 3.数组的交集,并集,差集(es6 Set)

const A = [1, 2, 3, 5, 7];
const B = [2, 4, 5, 6, 7];
const [ Sa, Sb ] = [new Set(A), new Set(B)];

// 并集
let union = Array.from(new Set([...A, ...B])); // [1, 2, 3, 5, 7, 4, 6]

// 交集 
let intersection = A.filter(v => Sb.has(v)); //[2, 5, 7]

// 差集
let diff = A.filter(v => !Sb.has(v)) // [1, 3]

// 补集
let complement = [...A.filter(v => !Sb.has(v)), ...B.filter(v => !Sa.has(v))] //1.3.4.6

# Array.prototype.reduce

    let arr =['1','2',[1,5,10],[1,2,5,[100,200,300]]];
    const flatten = (arr =[]) =>{
        return arr.reduce((a,b)=>a.concat(b),[]);
    } 
    flatten(arr); // ['1', '2', 1, 5, 10, 1, 2, 5, Array(3)]

# Array.prototype.concat

    let arr =['1','2',[1,5,10],[1,2,5,[100,200,300]]];
    const flatten = (arr =[]) =>{
        return [].concat( ...arr.map(x => Array.isArray(x) ? flatten(x) : x))
    }
    flatten(arr) // ['1', '2', 1, 5, 10, 1, 2, 5, 100, 200, 300]

# 双层循环

const unique = (arr) => {
    let i = 0, len = arr.length;
    for (i; i < len; i++) {
        for (let j = i + 1; j < len; j++) {
            if (arr[i] === arr[j] || (Object.is(arr[i], NaN) && Object.is(arr[j], NaN))) {
                arr.splice(j, 1)
                len-- // 减少循环次数提高性能
                j-- // 保证j的值自加后不变
            }
        }
    }
    return arr;
} //
unique(arr); //[1, 2, 'abc', true, false, undefined, NaN]
  • isNaN:判断的数据不准确,undefined的结果也是true;
  • NaN不能和NaN直接进行比较

# includes()

function unique(arr) {
  const newArr = []
  arr.forEach(item => {
    if (!newArr.includes(item)) {
      newArr.push(item)
    }
  })
  return newArr
}

const result = unique(arr) // [ 1, 2, 'abc', true, false, undefined, NaN ]

const unique =  arr => {
    const newArr = [];
    const obj = {};

        obj[item] = true
        }

}
