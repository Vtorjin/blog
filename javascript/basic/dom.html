# js常用功能- DOM

# 客户端下载图片

/**
 *@param  {src:string} 图片地址  
 *@param  {name:string}  自定义文件名称 
 */
const downImageFile = (src, name) => {
    let image = new Image();
    // 解决跨域 Canvas 污染问题
    image.setAttribute("crossOrigin", "anonymous");
    image.onload = function () {
        let canvas = document.createElement("canvas");
        canvas.width = image.width;
        canvas.height = image.height;
        let context = canvas.getContext("2d");
        context.drawImage(image, 0, 0, image.width, image.height);
        let url = canvas.toDataURL("image/png"); //得到图片的base64编码数据
        let a = document.createElement("a"); // 生成一个a元素
        let event = new MouseEvent("click"); // 创建一个单击事件
        a.download = name || "photo"; // 设置图片名称
        a.href = url; // 将生成的URL设置为a.href属性
        a.dispatchEvent(event); // 触发a的单击事件
        canvas = null;
        a = null;
        event = null;
    };
    image.src = src;
}

downImageFile('http://www.sikiedu.com/files/course/2021/10-14/17192914e765627633.png','pag.jpg');

注意: 某些图片的地址没有处理跨域,该方法是下载不了图片的.

# 客户端下载文本文件

const downloadTxtFile = (fileName,content) => {
    // 它适用于所有支持 HTML5 的浏览器，因为它使用了 <a> 元素的下载属性：
    let element = document.createElement("a");
    // Blob 是一种可以存储二进制数据的数据类型
    // 根据要保存的文件，它可以有不同的值
    const blob = new Blob([content], { type: "plain/text" });
    // createObjectURL() 静态方法创建一个 DOMString，其中包含一个 URL，该 URL 表示参数中给定的对象。
    const fileUrl = URL.createObjectURL(blob);
    // setAttribute() 设置指定元素的属性值。
    element.setAttribute("href", fileUrl); // 文件位置
    element.setAttribute("download", fileName); // 文件名
    element.style.display = "none";
    // 使用 appendChild() 方法将一个节点附加到指定父节点的子节点列表的末尾处
    document.body.appendChild(element);
    element.click();
    // Node 接口的 removeChild() 方法从 DOM 中移除一个子节点并返回移除的节点
    document.body.removeChild(element);
}

downloadTxtFile('file.txt',"这里是一段测试的文字!");
downloadTxtFile('key.key',"m3u8文件加密的key!");

# 获取元素中的所有图像

//includeDuplicates表示是否重复 
const getImages = (el, includeDuplicates = false) => {
  const images = [...el.getElementsByTagName('img')].map(img => img.getAttribute('src'));
  return includeDuplicates ? images : [...new Set(images)];
};

getImages(document, true); // ['image1.jpg', 'image2.png', 'image1.png', '...']
getImages(document, false); // ['image1.jpg', 'image2.png', '...']

# 滚动到顶部

最简单的滚动元素的方法是使用scrollIntoView方法。添加行为。"smooth "来实现平滑的滚动动画。

const scrollToTop = (element)  => element.scrollIntoView({ behavior: "smooth", block: "start" });
scrollToTop(document.body)

注:

  • beahavior有auto和smooth两种,
  • block定义垂直的方向,有start,center, end,nearest
  • inline定义水平滑动方向 start,center, end,nearest

# canvas压缩图片,并转为base64格式

/***
 * @params {string} url: 实现图片文件;
 * @params {object} config: 压缩图片的配置
 * */ 
 const compressAndConvertImage = (url, config = {}) => {
      return new Promise((resolve, reject) => {
        let canvas = document.createElement('canvas'),
          ctx = canvas.getContext('2d'),
          img = new Image();

        //开启cors功能，避免toBlob()、toDataURL()和getImageData()方法跨域
        img.setAttribute('crossOrigin', 'Anonymous');

        img.src = url;
        img.onload = () => {
          let width, height;
          //有百分比
          if (config.rate && typeof config.rate === 'number') {
            width = img.width * config.rate;
            height = img.height * config.rate;
          } else {
            // 没有压缩比率安装宽度进行压缩
            width = config.width || img.width;
            height = (width / img.width * img.height) || img.height;
          }
          // 设置canvas宽高
          canvas.width = width;
          canvas.height = height;
          //转换为base64Url quality: 

          //设置背景填充颜色以及绘制图片宽高
          ctx.fillStyle = '#fff';
          ctx.fillRect(0, 0, canvas.width, canvas.height);
          ctx.drawImage(img, 0, 0, width, height);

          // 获取base64地址
          const dataUrl = canvas.toDataURL(config.type || 'image/jpeg', { quality: config.quality || 0.7 });
          canvas = null;
         
         //将得到的base64Url返回出去 -- 不喜欢回调的可以用Promise进行处理
          resolve(dataUrl)
          //document.querySelector('img').src = dataUrl;
        }
      })
    }

  compressAndConvertImage('https://www.vicray.cn/images/niagara.jpeg',{width:400}).then(url => console.log(url));

注意: toDataUrl(type)的执行顺序需要放在背景填充色之后,否则得到jpeg格式的图片一直是黑色的(png的默认颜色是透明的).

config的可配置对象

  • width: 压缩后的宽度(填写宽度之后,会自动计算高度);
  • rate: 设置图片的整体缩放比(0-1之间);
  • quality: 图片整体的压缩质量

# base64地址转为blob对象

//参数: base64地址 
    const convertBaseUrlToBlob = (baseurl) => {
      let byteString;
      if (baseurl?.split(',')[0]?.indexOf('base64') !== -1) {
        byteString = atob(baseurl.split(',')[1]); //base64解码
      } else {
        byteString = unescape(baseurl.split(',')[1]);
      }
      if (!byteString) return null;

      //[data:image/png;base64,xxxx] =>  [data,xxxxx] =>  [image/png or image/jpeg, xxxxxx]; => image/png or image/jpeg
      const mimeString = baseurl?.split(',')[0]?.split(':')[1]?.split(';')[0];  //获取url的类型
      if (!mimeString) return null;

      let [imgArray, i, len] = [new Uint8Array(byteString.length), 0, byteString.length];

      for (i; i < len; i++) {
        imgArray[i] = byteString.charCodeAt(i);
      }
      
      return new Blob(imgArray, { type: mimeString });  //Blob {size: 20837, type: 'image/jpeg'}
    }

# base64转file对象

const convertBaseUrlToFile = (dataUrl, filename = 'image/png') => {
    try {
        let arr = dataUrl.split(','),
            mime = arr[0].match(/:(.*?);/)[1],
            bstr = atob(arr[1]),
            n = bstr.length,
            u8arr = new Uint8Array(n);
        while (n--) {
            u8arr[n] = bstr.charCodeAt(n);
        }
        return new File([u8arr], filename, { type: mime });
    } catch (e) {
        console.log(e);
        return null
    }
}

# 将blob对象转为base64格式


/**
 *@param {Blob} blob:blob对象  
 *@return {Promise}
 *@description 传入的是blob对象,而不是以blob:开头的地址 
 **/

const convertBlobToBase = (blob) => {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.readAsDataURL(blob);
        reader.onload = (dataUrl) => {
            resolve(dataUrl);
        }
    })
}

# 网页开启护眼模式

注: 建议在浏览器中安装油猴插件(Tampermonkey)然后添加此段代码

const whiteList =['loaded']; //dom白名单
const theme = new Date().getHours() > 20 ? "#5f6a7d" :"#C7EDCC"; //设置背景:晚间用淡黑色,白天用淡绿色
const font = "#333"; //字体颜色
const openEyeProtect = function(dom){
    if(dom.childNodes&&dom.childNodes.length){
        dom.childNodes.forEach(it=>{
            if(!whiteList.includes(it.className)){
                it.style && (it.style.background= theme)&&(it.style.color= font )
            }
            if(it.childNodes){
                openEyeProtect(it);
            }
        })
    }else{
        return;
    }
}

openEyeProtect(document.querySelector('body'))
