const getValueType = (value) => {
    if(value === null) return value + ''; 

    if(typeof value !== 'object') return typeof value;

    let valueClass = Object.prototype.toString.call(value),
        type = valueClass.split(' ')[1].split('');
        type.pop();
        
    return type.join('').toLowerCase();
}

# 闭包应用-防抖

防抖(debounce):任务频繁触发的情况下，只有任务触发的间隔超过指定间隔的时候，任务才会执行 应用场景

  • 输入框在用户完成框输入条件后间隔显示结果，在输入期间不显示结果。
  • button按钮点击事件、点赞。
  • 窗口resize事件。
  • 创建订单等不可同一时间内频繁进行操作的场景；
const debounce = (fn, delay = 200) => {
    //创建一个标记用来存放定时器的返回值
    let timeout = null;
    return function () {
        var that = this;
        //每次当用户点击/输入的时候，把前一个定时器清除
        clearTimeout(timeout);
        //然后创建一个新的 setTimeout，
        timeout = setTimeout(function(){
            fn.call(that, arguments);
        }, delay || 200);
    };
}

# 闭包应用-节流

节流(throttle)：控制流量，单位时间内事件只能触发一次,节流重在开关锁timer=null;

应用场景：

  • 窗口scroll 事件，每隔一秒计算一次位置信息等
  • 浏览器播放事件，每隔一秒计算一次进度信息等
const throttle = (f, delay) => {
    let timer;
    return function(){
        var that = this;
        if(timer) return;
        timer = setTimeout(function(){
            fn.call(this,...arguments);
            timer = null;
        },delay)
    }   
}

# 自定义-call

Function.prototype.myCall =  function(thisArg, ...args){
    //获取要被执行的函数
    var fn = this;
    //将当前的指针转成对象(防止它传入的是非对象类型的参数)
    thisArg = thisArg ? Object(thisArg) : window;
    //为调用的myCall的函数绑定this(利用了js的隐式绑定)
    thisArg.fn =  fn;
    //调用需要被执行的函数
    const result = thisArg.fn(...args);
    delete thisArg.fn;
    //返回执行结果
    return result;
} 

const result = sum.myCall('aaa', 10, 20)
console.log(result)

function sum(num1, num2) {
    console.log('sum函数被执行了', this)
    return num1 + num2
}

const result2 = sum.call('aaa', 20, 30)
console.log(result2)

# 自定义-apply

Function.prototype.apply = function (thisArg,argArr){
    // 1. 获取调用myApply的函数
    const fn  = this;
    // 2. 将thisArg转成对象类型，否则后续无法通过thisArg.fn绑定this
    thisArg = (thisArg !== null && thisArg !== undefined) ? Object(thisArg) : window
    thisArg.fn = fn;
    // 3. 对argArr进行处理，确保它有传值并且为数组
    argArr = argArr ? argArr : [];
    // 4. 执行函数
    const result = thisArg.fn(...argArr);
    delete thisArg.fn;
    // 5. 返回结果
    return result;
}

# 自定义-bind

Function.prototype.myBind = function(thisArg,...argArr){
    // 1.获取真实需要调用的函数
    var fn = this;
    // 2.处理thisArg不传值的情况
    thisArg = (thisArg !== null && thisArg !== undefined) ? Object(thisArg) : window;

    var proxyFn = function(...args){
        // 3.绑定this
        thisArg.fn = fn;
        // 4.合并参数
        const finalArgArr = [...argArr, ...args];
        // 5.调用函数
        const result = thisArg.fn(...finalArgArr);
        delete thisArg.fn;
        return result;
    } 

    return proxyFn;
}

// 示例: 分页组件页码改变后重新获取数据
// create Event
const createEvent = (eventName,message)=>{
    return new CustomEvent(eventName,{detail:message});
}

// listen event
const listenEvent = function(eventName){
    window.addEventListener(eventName, function({detail}){
        console.log('your event received a new params', detail)
        // another code xxxx
    })
}
// patch event
const dispatch = (event) => {
    window.dispatchEvent && window.dispatchEvent(event);
}
// delete custom event
const removeEvent = (eventName) => {
    window.removeEventListener && window.removeEventListener(eventName)
}

// first: listen event
listenEvent('updateTableData');

// next: create event
const initEvent = createEvent('updateTableData', {searchValue:"", pageNum: 1, pageSize: 100});

// last: dispatch event
dispatch(initEvent);

// if some params had been changed, recreate event.
const changeEvent = createEvent('updateTableData',{searchValue:"new VAL", pageNum : 5 ,})  

// next: dispatch
dispatch(changeEvent);

# 字节Mb,Gb,Tb转换

const convertKb = (byte, unit = 'kb') => {
    unit = String(unit).toLowerCase();
    switch (unit) {
        case 'bit':
            return byte * 8 + 'b';
        case "byte":
            return byte + 'B';
        case "kb":
            return (byte / 1024) + 'Kb';    
        case "mb":
            return (byte / 1024 ** 2).toFixed(2)+ 'Mb';    
        case "gb":
            return (byte / 1024 ** 3).toFixed(2)+ 'Gb';
        case "tb":
            return (byte / 1024 ** 4).toFixed(2)+ 'Tb';    
        default:
            console.warn('该单位不支持!')
    }
}

// window.performance.memory下都是以'B'为单位
convertKb(window.performance.memory.jsHeapSizeLimit,'Mb'); //上下文内可用堆的最大体积
convertKb(window.performance.memory.totalJSHeapSize,'Mb'); // 已分配的堆体积
convertKb(window.performance.memory.usedJSHeapSize,'Mb'); // 当前 JS 堆活跃段（segment）的体积
