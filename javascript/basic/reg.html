# 获取字符串的指定的完整单词

const getSpecifyStr = (str,word) => {
   try{
       const reg = eval(`/\\b${word}\\b/g`);
       return reg.exec(str)?.[0]
   }catch(e){
       return null
   }
}

getSpecifyStr('its called jenny','call') // undefined;
getSpecifyStr('its called jenny','jenny') //jenny;
getSpecifyStr('its called jenny','its  called') //undefined

# 匹配日期格式

适用于不支持es9版本的浏览器

//适用于文本匹配；不适合准确值校验
const checkDate = (val) => {
    const reg = /^\d{4}-([1-9]|[1][0-2]|[0][1-9])-([0][1-9]|[1-2][0-9]|[3][01])/g;
    return reg.exec(val)?.[0]
}
checkDate('2022-12-02'); //'2022-12-02'
checkDate('2022-1-2'); //'2022-1-2'

# 提取Img,Video元素的url地址

/**
 *@val {string} 文本地址 
 *@source {string} src || href
 *@fileType {string} mp4||mp3||m3m8||jpg  
 * 
 */
const getUrl = (val,source = 'src',fileType = '">') => {
   try{
       const reg = eval(`/(?<=(${source}="))\\S*(?=(${fileType}))/g`);
       const res = reg.exec(val);
       return res ? `${res.shift()}${fileType}` : res;
   }catch(e){
     return null
   }
}

const url = '<img style="line-height: 1.5; background: rgb(199, 237, 204); color: rgb(51, 51, 51);" src="https://images2015.cnblogs.com/blog/290384/201605/290384-20160505183727232-219430292.png" alt="" class="medium-zoom-image">';

getUrl(url,'src','png');
