class PSModel{
    static instance;
    static getInstance(){
        if(!this.instance){
            this.instance = new PSModel();
        }
        return this.instance;
    };
    clients;
    constructor(){
        this.clients = {};
    }
    sub(key,fn){
        (this.clients[key] || (this.clients[key] = []) ).push(fn);
    }
    pub(){
        let key = Array.prototype.shift.call(arguments),
			  len = this.clients[key]?.length;
        if(!this.clients[key] || !len) return;
        const client = this.clients[key];
        while(len--){
            client[len].apply(this,arguments);
        }
    }
}

let dataCenter = ({data, subTag, dataKey, selector}) => {
    let [el, val] = [document.querySelector(selector),''];
    Object.defineProperty(data, dataKey, {
        get(){
            return val;
        },
        set(v){
			val = v;
            PSModel.getInstance().pub(subTag, v);
        }
    });

    PSModel.getInstance().sub(subTag,(newVal)=>{
        el && el.innerText = newVal;
    })
}

# Proxy

target: The target parameter indicates the target object to be intercepted.

handle: The handle parameter is also an object used to customized the interception behavior.

const user = {};
// three handler hooks in Proxy. 
const handler = {
          get(target,key,receiver){
                return Reflect.get(target,key, receiver)
          },
          set(target,prop,value,receiver){
            user[prop] = value;
            return Reflect.set(target, prop, value, receiver)
          },
          deleteProperty(target,prop){
            console.log('I am triggered the delete hook,prop is' + prop)
              return Reflect.deleteProperty(target,prop);
          },
};
const proxy = new Proxy(user,handler)
console.log(proxy.name); //undefined
proxy.name = 'funny'; // funny
console.log(proxy.name);
delete proxy.name; //true

# Reflect

main function: optimize some operation methods of object and reasonably and return the results returned by the object operation.

let obj = {
			name:'zhangsan',
			age:'18',
			func:function(){
				console.log(test);
			},
			
		}
Object.defineProperty(obj,'job',{
    value:'programer',
    enumerable:false
})

console.log(Object.keys(obj)); //(3) ["name", "age", "func"]
console.log(Reflect.ownKeys(obj)); //(4) ['name', 'age', 'func', 'job']
  • Reflect.get: roughly equivalent to the Object.defineProperty,only difference is its return a boolean.
  • Reflect.set(target,property,value): this methods allows you to set attribute and value into an object,and return a boolean.
  • Reflect.deleteProperty: delete an attribute in an object。
  • Reflect.defineProperty:
  • Reflect.ownKeys(target):Object.keys(obj) returns an array includes all enumerable attributes and ownKeys return all attributes.

const reactive = (target = {}, subTag, selector) => {
    if(typeof target !== "object" || target === null ){
        return target;
    }
    const el = document.querySelector(selector);
   
    const handler  = {
        get(target,key,receiver){
             const res = Reflect.get(target,key, receiver);
             return typeof res==='object'? reactive(res) : res ; 
        },
        set(target,key,value,receiver){
            if(target[key] === value) return true;
                //some keys had existed in target
            if(Reflect.ownKeys(target).includes(key)){
                 PSModel.getInstance().pub(subTag, value);
            }else{
                 //the new keys to add to target
                 console.log("i'm the latest key:" + prop)
            }
            const res = Reflect.set(target,key,value,receiver);
            return res;
        },

			if(Reflect.ownKeys(target).includes(prop)){
