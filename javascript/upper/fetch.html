# es6封装Fetch实现不同请求类型方式

以class的方式封装fetch,实现多个请求方式(get,post,put,delete)的实现,以及实现文件上传,取消请求的操作.

// let baseUrl = "https://www.vicray.cn:3886";

//获取数据
class FetchHttp {
    static instance;
    static getInstance() {
        if (this.instance == null) {
            this.instance = new FetchHttp()
        }
        return this.instance;
    }
    static baseUrl;

    /**
    * get 请求
    * @param {string} url 请求路径 
    * @param {object} params 请求参数
    * @returns {Promise}
    * @description get请求只设置两个参数,一个是请求路径,一个是请求的参数对象 
    */
    get(url, params = {}) {
        const param = {};
        param.method = "GET";
        Object.keys(params).length ? (param.params = params) : null;
        return this.sendRequest(url, param)
    }

    
    /**
     * @param {string} url 请求路径
     * @param {object} data data内容格式包含:{type:"json or urlencoded",data:"请求参数", headers:{'请求头参数配置'}}
     * @return {Promise}
     * @description post请求的类型type的可以设置'type'和'json'两种,并且请求参数放在body里面
     **/
    post(url, data = { body: {} }) {
        data.method = 'POST';
        data.headers = {};
        return this.sendRequest(url, data)
    }

    //同上
    put(url, data = { body: {} }) {
        data.method = 'PUT';
        data.headers = {};
        return this.sendRequest(url, data)
    }

    // 同get请求
    delete(url, params = {}) {
        const param = {};
        param.method = 'DELETE';
        param.params = params;
        return this.sendRequest(url, param)
    }

    //发送请求
    sendRequest(url, options = {}) {
        // 设置请求路径
        let fullUrl = `${FetchHttp.baseUrl || ''}${url}`;

        // 对get以及参数进行处理
        !options.hasOwnProperty('method') ? (options.method = 'GET') : null;

        if (options.hasOwnProperty('params') && ['GET', 'DELETE', 'HEAD', 'OPTIONS'].includes(options.method.toUpperCase())) {
            let str = stringifyObj(options.params);
            fullUrl += `?${str}`
            // params不是fetch中自带的有效参数，fetch不支持该参数，需要在发送请求前将其删除
            delete options.params;
        }

        //对post以及参数进行处理
        if (['POST', 'PUT'].includes(options.method.toUpperCase())) {
            // 读取传入的数据格式类型参数type，如果没有传入type，默认为urlencoded格式
            options.type = !options.type ? 'urlencoded' : options.type;

            if (options.type === 'urlencoded') {
                options.headers['Content-type'] = 'application/x-www-form-urlencoded';
                options.body = stringifyObj(options.body);
            }

            if (options.type === 'json') {
                options.headers['Content-type'] = 'application/json';
                options.body = JSON.stringify(options.body);
            }
        }


        //合并配置项,Object.assign会覆盖因此提前
        options = Object.assign({
            //允许跨域携带资源凭证 include: 无论是否同源都可以; same-origin:同源可以; omit:都拒绝
            // credentials: "include",
            headers: {}
        }, options)


        options.headers.Accept = 'application/json;charset=utf-8'; //可以根据需求填写
        localStorage.getItem('token') && (options.headers.Authorization = localStorage.getItem('token')); //携带token

        return fetch(fullUrl, options)
            .then(response => {

                options = null;
                if (!(200 < response.status < 300)) {
                    switch (response.status) {
                        case 401:
                            // 未登录
                            console.log('未登录的操作!');
                            return;
                        case 403:
                            console.log('服务器拒绝的操作!')
                            return;
                        case 404:
                            console.log('页面未找到的操作!')
                            return;
                        default:
                            console.log('默认的操作')
                    }
                }
                return response.json();
            })
            .then(res => {
                console.log(res);
            })
            .catch(error => {
                options = null;
                if (!window.navigator.onLine) {
                    //断网处理
                }
                //什么都没有，返回错误
                return Promise.reject(error)
            })
    }
}

//将对象的键值对用"="的形式进行拼接 
function stringifyObj(obj = {}) {
    return JSON.stringify(obj).replaceAll(/[,":{}]/g, function (a) {
        switch (a) {

            case ":":
                return "=";
            case ",":
                return "&";

            case "{":
            case "}":
            case '"':
                return "";

            default:
                return a
        }
    })
}
// 第一步 设置默认地址
FetchHttp.baseUrl = "https://www.vicray.cn:3886";

// 第二步 调用接口
FetchHttp.getInstance().get('/test', { 'textArea': "get请求" })
FetchHttp.getInstance().post('/test', { body: { name: "上班使我快乐", 'info': "造孽啊" }, type: "json" })
FetchHttp.getInstance().put('/test', { body: 'name=上班使我快乐&info=日常码畜', type: "urlencoded" })
FetchHttp.getInstance().delete('/test', { 'textArea': "delete请求" })

注意:对于请求的参数格式(headers下的{content-type:xxxxxx})需要提前和后端同学约定好,以免造成后端数据无法获取的尴尬.
