# es6实现EventBus

去年参加了一个微前端项目乾坤,该应用下存在若干的子应用,不同子应用之间存在数据通信的需求,但是子应用之间的是相互独立的,为了能够使不同独立的应用之间进行通信,当时写了一个eventBus放置在框架的SDK中，供其他应用进行使用.

class EventBus{
    static instance;
    static getInstance(){
        if(this.instance == null ){
            this.instance  = new  EventBus();
        }
        return this.instance;
    }
    constructor(){
        this.events = {};
    }

    registerEvent(id, listener, tag){
        if(typeof id !=='string' || typeof listener !== 'function' ){
            throw new Error('订阅事件格式错误!')
        }
        if(!listener.name && !tag){
            throw new Error('事件订阅匿名函数需要添加唯一标识!');
        }
        if(!this.events[id]){
            this.events[id] = {};
        }
        
        let listenerName = listener.name? listener.name : tag;
        if(this.events[id][listenerName]){
            throw new Error('重复订阅事件!');
        }
        this.events[id][listenerName] = listener;
    }

    postEvent(id,...message){
       if (id && this.events[id]) {
            Object.values(this.events[id]).forEach(listener => {
                listener.call(this, ...message);
            })
        }    
    }

    }
}

// 使用示例
EventBus.getInstance().registerEvent('deleteJoin', (msg) => { console.log('one', msg) }, 'test')
EventBus.getInstance().registerEvent('deleteJoin', function test2(msg){ console.log('two', msg) }, 'test2')
