# function makeMap

/**
 * @param {str:string} str: such as 'slot,component,key,ref,slot-scope, is'
 * @param {expectsLowerCase:boolean} expectsLowerCase: a boolean value and it's unnecessary！
 * @return {object} return a object that keys limited case and set its value as true ;
 * @description: define some value and easy to use at a later time 
 **/ 
export function makeMap (str: string,expectsLowerCase?: boolean): (key: string) => true | void {
    const map = Object.create(null)
    //list is an array which consists of string elements. 
    const list: Array<string> = str.split(',')
    for (let i = 0; i < list.length; i++) {
        map[list[i]] = true
    }
    return expectsLowerCase ? val => map[val.toLowerCase()] : val => map[val]
}
// exp: makeMap('key,ref,slot,slot-scope,is') => f (val) {return map[val]}
  • differences between Object.create(null) and {}:

Object.create(null) which creates a absolute empty object,it has no attributes or function on prototype chain (such as toString(), hasOwnProperty()).

# cached,camelize, capitalize, hyphenate

/**
 * @param {function} fn: a pure function
 * @description  Create a cached version of a pure function.
 **/

export function cached <F:Function> (fn:F):F{
    const cache = Object.create(null);
    return (function cacheFn(str:string){
        const hit = cache[str];
        return hit || (cache[str] = fn(str));
    }:any); // used closure;
}

/**
 * @param {str:string} str: a hyphen-delimited string
 * @return {a Camelized string}
 * @description: Camelize a hyphen-delimited string(such as border-radius => borderRadius)
 **/ 
const camelizeRE = /-(\w)/g;
export const camelize = cached(str:string):string =>{
    return str.replace(camelizeRE,(_,c)=> c ? c.toUpperCase() : '');
}

//Capitalize a string.
export const capitalize = cached((str: string): string => {
  return str.charAt(0).toUpperCase() + str.slice(1)
})

//Hyphenate a camelCase string.
const hyphenateRE = /\B([A-Z])/g;
export const hyphenate = cached((str:string) :string =>{
    return str.replace(hyphenateRE,'-$1').toLowerCase();
})
  • why need cached function?

if there is a camelizeRE function with complex logic and need take a lot of time to execute, and sometimes I have to execute it so many times with the same params. Actually, it will get the same execute result.so when the code is executed repeatly that will takes user's many time. the code the function cached is to read data directly form memory and don't take a long time. that's called trade time from space.

# function looseEqual

/**
 *@return {boolean}
 *@description: Check if two values are loosely equal. if they are plain objects, do they have the same shape?
 **/ 
const looseEqual = (a:any, b:any):boolean => {
    if(a === b) return true;
    const isObjectA:boolean = isObject(a); // the same filePath
    const isObjectB:boolean = isObject(b);
    if (isObjectA && isObjectB) {
        try {
        const isArrayA = Array.isArray(a)
        const isArrayB = Array.isArray(b)
        if (isArrayA && isArrayB) {
            return a.length === b.length && a.every((e, i) => {
            return looseEqual(e, b[i])
            })
        } else if (a instanceof Date && b instanceof Date) {
            return a.getTime() === b.getTime()
        } else if (!isArrayA && !isArrayB) {
            const keysA = Object.keys(a)
            const keysB = Object.keys(b)
            return keysA.length === keysB.length && keysA.every(key => {
            return looseEqual(a[key], b[key])
            })
        } else {
            /* istanbul ignore next */
            return false
        }
        } catch (e) {
        /* istanbul ignore next */
        return false
        }
    } else if (!isObjectA && !isObjectB) {
        return String(a) === String(b)
    } else {
        return false
    }
}

                });
            case rowTypeA == 'Object':

                const keysB = Object.keys(b);
                return keysA.length === keysB.length && keysA.every(key => {

# Array-like to an Real Array

/**
 *@description convert an Array-like object to a real Array
 **/ 
const toArray = (list:any, start:number):Array<any>  => {
    start =  start || 0;
    let i = list.length - start;
    const ret:Array<any> = new Array(i);
    while(i--){
        ret[i] = list[i + start];
    }
    return ret;
}
