# 油猴爬虫系列: 批量下载图片

对于图片网站进行下载图片时，一个一个进行下载保存太麻烦，因此想对批量图片下载功能记录一篇。

# Html部分(浏览器环境)

  • 图片地址列表

一般的图片网站都是将图片列表存放于某个容器中，比起考虑到一些性能问题，往往会采用懒加载的处理方式，因此在获取想要下载的地址时，可以考虑到img上的 src 或者 dataset-src;

# 无分页版本

// 快速获取图片列表
const  getUrlList = (el = "" ) => {
    // 自己用的就不去做一些参数判断
    const imgs = document.querySelectorAll(`${el} img`);

    var list = Array.prototype.slice.call(imgs).map(img => {
        return img.src || img.dataset && img.dataset.src;
    })

    return list;
} 

// 想要将文本放在本地时直接可以进行复制
document.body.innerHTML = getUrlList('#ImageContainer').join(',<br>');

获取到图片地址列表后，可以将数据放到本地文件上

# 分页版本

如果有涉及到分页的时候，可以用一些工具，例如油猴脚本，Jquery,进行快速访问指定页码的html文件，方便爬取数据。

// 目录页操作
const getPages = (el = '') =>  Array.prototype.slice.call(document.querySelectorAll(`${el} a`)).map(a => {
    return a.href;
})


// 请求页面收集图片列表
let collect = [];

// 参数1: 页码列表, 参数2： 每个页面上存放图片列表的根容器名称
const request = async(list = [], el = "#post_content") => {
    var reg = /<body(([\s\S])*?)>?<\/body>/g;
    // 本地保存数据可以用 for... of  await 串行的方式  缺点就是页面的href太多就会等待太久
    for(const page of list){
        const res = await $.ajax({
            url:page,
            success(data){
                let body = data.match(reg);
                body && $(body[0]).find(`${el} img`).each((i,img)=>{
                    //有些时候，图片的sr没有添加请求协议，这里可自行修改
                    var url = ($(img).attr('src') ||  $(img).attr('src'));  
                    url && collect.push(url);
                });
                
            },
        })
        window.stop(); //停止资源加载
      
    }
}

// 示例网站 https://www.vcg.com/sets/505554717
request(getPages('#imageContent'), ".relativePosition")
.then(() => {
    // 收集到图片之后，可以选择交发送请求给服务端处理，或者将列表保存在本地
    console.log(collect,'收集到的图片地址列表');    
    // 这里我选择的是本地保存
    document.body.innerHTML = collect.join(',<br>')
});

# Node部分

  • 需要准备的包
var request = require('request')
var fs = require('fs')
var path = require('path')
var https = require("https"); 
// 如果所有的操作都是在node环境下执行，就需要引入 cheerio
// var io = require("cheerio");
  • 下载单张图片
function saveImage(url, fileName) {
    var rq = https.get(url, function (req, res) {
        var imgData = '';
        req.on('data', function (chunk) {
            imgData += chunk;
        })
        req.setEncoding('binary');   
        req.on('end', function (e) {    
            fs.writeFile(fileName, imgData, 'binary', function (err) {  
                console.log('图片下载成功', fileName);
            })
        });
    })
    // 监听下载错误
    rq.on('error', function () {
        console.log('失败了！ 重新试试呢') 
        saveImage(url, fileName); //重新进行下载
    })          
})
  • 可优化部分

    }else{
        console.log('存在,获取下一张')

req.on('end', function (e) {

        saveImage('nextUrl', 'nextFileName');

在根目录创建image文件夹后可以使用，如果要下载的图片地址中有文件夹的路径标识，还可以自动去根据标识创建文件夹,这里就不做处理了

// crawler_downloadImg.js

var request = require('request')
var fs = require('fs')
var path = require('path')
var https = require("https");
var list = require('./list'); //本地保存的图片地址列表


var l = 0, 
    len = list.length
    url = list[0];;

function getFileName(url = ''){
    var name = url.split('/');
    return name[name.length-1];
}

function saveImage(url, fileName) {
    if (!url || l >= len) {
        console.log('已下载完毕！');
        return;
    }
    fs.access(fileName, (e) => {
        if (!e) {
            console.log('已经存在了', l - 1)
            l++;
            l < len && saveImage(list[l], path.join(__dirname, "image/" + getFileName(list[l])));
            return;
        }
        // 请求地址
        var rq = https.get(url, function (req, res) {
            var imgData = '';
            req.on('data', function (chunk) {
                imgData += chunk;
            })
            req.setEncoding('binary');
            req.on('end', function (e) {
                fs.writeFile(fileName, imgData, 'binary', function (err) {
                    // 递归进入下一条数据
                    l++;
                    saveImage(list[l], path.join(__dirname, "image/" + getFileName(list[l])))
                })
            });
        })
        // 监听下载错误
        rq.on('error', function () {
            console.log('失败了! 重新试试呢')
            saveImage(url, fileName);
        })

    })
}
