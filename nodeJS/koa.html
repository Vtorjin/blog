# Koa框架托管静态资源并开启https服务

//app.js 
const Koa = require('koa');
const sslify = require('koa-sslify').default;
const path = require('path');
const https = require('https');
const fs = require('fs');
const cors = require('cors');
const app = new Koa();

// 配置ssl证书的key和cert
const httpsOption = {
    key:fs.readFileSync('your key address.key'),
    cert:fs.readFileSync('yoru pem address.key')
}

//使用ssl
app.use(sslify());
//开启跨域
app.use(cors());

//托管静态资源
app.use(require('koa-static')(path.join(__dirname,'your project folder')))

//不同于express, koa的createSever第二个参数是一个回调,直接传app就会报错
const httpsSever = https.createSever(httpsOption, app.callback()); 

httpsSever.listen('port',(err)=>{
    console.log(`port${port}启动成功`)
})
